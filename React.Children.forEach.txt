React.Children.forEach là mọt tính năng khác có thể được dùng trên this.props.children. Nó tương
tự với React.Children.map function ngoại trừ nó không trả về một object

React.Children.forEach(children, myFn[, context])

Ví dụ:

var MyComponent = React.createClass({
  render: function() {
    React.Children.forEach(this.props.children, function(child) {
      console.log(child);
    });
    return (
      <div>{this.props.name}</div>
    );
  }
});

React.render(<MyComponent name="frodo">
  <p key="firsty">a Child</p>
  <p key="2">another</p>
</MyComponent>, document.getElementById('container'));
